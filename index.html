      <script>
      let allTransactions = [];
      let currentRecordCount = 0;
      const defaultConfig = {
      app_title: "Catatan Keuangan",
      income_label: "Pemasukan",
      expense_label: "Pengeluaran",
      background_color: "#667eea",
      card_color: "#ffffff",
      text_color: "#1f2937",
      primary_button_color: "#9333ea",
      secondary_button_color: "#ef4444"
      };
      const dataHandler = {
      onDataChanged(data) {
      allTransactions = data.sort((a, b) => new Date(b.date) - new Date(a.date));
      currentRecordCount = data.length;
      if (currentRecordCount >= 999) {
      document.getElementById('limit-warning').classList.remove('hidden');
      } else {
      document.getElementById('limit-warning').classList.add('hidden');
      }
      renderTransactions();
      updateSummary();
      }
      };
      async function onConfigChange(config) {
      const appTitle = config.app_title || defaultConfig.app_title;
      document.getElementById('app-title').textContent = appTitle;
      const typeSelect = document.getElementById('type-select');
      const incomeLabel = config.income_label || defaultConfig.income_label;
      const expenseLabel = config.expense_label || defaultConfig.expense_label;
      typeSelect.options[0].text = incomeLabel;
      typeSelect.options[1].text = expenseLabel;
      }
      function formatCurrency(amount) {
      return new Intl.NumberFormat('id-ID', {
      style: 'currency',
      currency: 'IDR',
      minimumFractionDigits: 0
      }).format(amount);
      }
      function formatDate(dateString) {
      const date = new Date(dateString);
      return new Intl.DateTimeFormat('id-ID', {
      day: 'numeric',
      month: 'long',
      year: 'numeric'
      }).format(date);
      }
      function isToday(dateString) {
      const date = new Date(dateString);
      const today = new Date();
      return date.toDateString() === today.toDateString();
      }
      function isThisWeek(dateString) {
      const date = new Date(dateString);
      const today = new Date();
      const firstDayOfWeek = new Date(today.setDate(today.getDate() - today.getDay()));
      return date >= firstDayOfWeek;
      }
      function isThisMonth(dateString) {
      const date = new Date(dateString);
      const today = new Date();
      return date.getMonth() === today.getMonth() && date.getFullYear() === today.getFullYear();
      }
      function isThisYear(dateString) {
      const date = new Date(dateString);
      const today = new Date();
      return date.getFullYear() === today.getFullYear();
      }
      function updateSummary() {
      let dailyIncome = 0, dailyExpense = 0;
      let weeklyIncome = 0, weeklyExpense = 0;
      let monthlyIncome = 0, monthlyExpense = 0;
      let yearlyIncome = 0, yearlyExpense = 0;
      allTransactions.forEach(trans => {
      const amount = trans.amount;
      if (isToday(trans.date)) {
      if (trans.type === 'income') dailyIncome += amount;
      else dailyExpense += amount;
      }
      if (isThisWeek(trans.date)) {
      if (trans.type === 'income') weeklyIncome += amount;
      else weeklyExpense += amount;
      }
      if (isThisMonth(trans.date)) {
      if (trans.type === 'income') monthlyIncome += amount;
      else monthlyExpense += amount;
      }
      if (isThisYear(trans.date)) {
      if (trans.type === 'income') yearlyIncome += amount;
      else yearlyExpense += amount;
      }
      });
      document.getElementById('daily-income').textContent = formatCurrency(dailyIncome);
      document.getElementById('daily-expense').textContent = formatCurrency(dailyExpense);
      document.getElementById('daily-balance').textContent = Saldo: ${formatCurrency(dailyIncome - dailyExpense)};
      document.getElementById('weekly-income').textContent = formatCurrency(weeklyIncome);
      document.getElementById('weekly-expense').textContent = formatCurrency(weeklyExpense);
      document.getElementById('weekly-balance').textContent = Saldo: ${formatCurrency(weeklyIncome - weeklyExpense)};
      document.getElementById('monthly-income').textContent = formatCurrency(monthlyIncome);
      document.getElementById('monthly-expense').textContent = formatCurrency(monthlyExpense);
      document.getElementById('monthly-balance').textContent = Saldo: ${formatCurrency(monthlyIncome - monthlyExpense)};
      document.getElementById('yearly-income').textContent = formatCurrency(yearlyIncome);
      document.getElementById('yearly-expense').textContent = formatCurrency(yearlyExpense);
      document.getElementById('yearly-balance').textContent = Saldo: ${formatCurrency(yearlyIncome - yearlyExpense)};
      }
      function renderTransactions() {
      const container = document.getElementById('transactions-list');
      if (allTransactions.length === 0) {
      container.innerHTML = '<p class="text-gray-500 text-center py-8">Belum ada transaksi. Mulai tambahkan transaksi pertama Anda!</p>';
      return;
      }
      const existingItems = new Map(
      [...container.children].map(el => [el.dataset.transactionId, el])
      );
      allTransactions.forEach(trans => {
      if (existingItems.has(trans.__backendId)) {
      updateTransactionElement(existingItems.get(trans.__backendId), trans);
      existingItems.delete(trans.__backendId);
      } else {
      container.appendChild(createTransactionElement(trans));
      }
      });
      existingItems.forEach(el => el.remove());
      }
      function createTransactionElement(trans) {
      const div = document.createElement('div');
      div.dataset.transactionId = trans.__backendId;
      div.className = 'transaction-item flex items-center justify-between p-4 bg-gray-50 rounded-lg border border-gray-200';
      const typeColor = trans.type === 'income' ? 'text-green-600' : 'text-red-600';
      const typeIcon = trans.type === 'income' ? '↑' : '↓';
      div.innerHTML = `
      <div class="flex-1">
      <div class="flex items-center gap-2 mb-1">
      <span class="text-2xl ${typeColor}">${typeIcon}</span>
      <span class="font-semibold text-gray-800">${trans.category}</span>
      </div>
      <p class="text-sm text-gray-600">${trans.description || '-'}</p>
      <p class="text-xs text-gray-500 mt-1">${formatDate(trans.date)}</p>
      </div>
      <div class="text-right">
      <p class="text-xl font-bold ${typeColor}">${formatCurrency(trans.amount)}</p>
      <button class="delete-btn mt-2 px-3 py-1 bg-red-500 text-white text-sm rounded hover:bg-red-600 transition duration-200">
      Hapus
      </button>
      </div>
      `;
      const deleteBtn = div.querySelector('.delete-btn');
      deleteBtn.addEventListener('click', () => handleDelete(trans, deleteBtn));
      return div;
      }
      function updateTransactionElement(element, trans) {
      const typeColor = trans.type === 'income' ? 'text-green-600' : 'text-red-600';
      const typeIcon = trans.type === 'income' ? '↑' : '↓';
      element.innerHTML = `
      <div class="flex-1">
      <div class="flex items-center gap-2 mb-1">
      <span class="text-2xl ${typeColor}">${typeIcon}</span>
      <span class="font-semibold text-gray-800">${trans.category}</span>
      </div>
      <p class="text-sm text-gray-600">${trans.description || '-'}</p>
      <p class="text-xs text-gray-500 mt-1">${formatDate(trans.date)}</p>
      </div>
      <div class="text-right">
      <p class="text-xl font-bold ${typeColor}">${formatCurrency(trans.amount)}</p>
      <button class="delete-btn mt-2 px-3 py-1 bg-red-500 text-white text-sm rounded hover:bg-red-600 transition duration-200">
      Hapus
      </button>
      </div>
      `;
      const deleteBtn = element.querySelector('.delete-btn');
      deleteBtn.addEventListener('click', () => handleDelete(trans, deleteBtn));
      }
      async function handleDelete(trans, button) {
      button.disabled = true;
      button.textContent = 'Menghapus...';
      const result = await window.dataSdk.delete(trans);
      if (!result.isOk) {
      button.disabled = false;
      button.textContent = 'Hapus';
      alert('Gagal menghapus transaksi. Silakan coba lagi.');
      }
      }
      document.getElementById('transaction-form').addEventListener('submit', async (e) => {
      e.preventDefault();
      if (currentRecordCount >= 999) {
      document.getElementById('limit-warning').classList.remove('hidden');
      return;
      }
      const submitBtn = document.getElementById('submit-btn');
      const submitText = document.getElementById('submit-text');
      const submitSpinner = document.getElementById('submit-spinner');
      submitBtn.disabled = true;
      submitText.textContent = 'Menyimpan...';
      submitSpinner.classList.remove('hidden');
      const transaction = {
      id: Date.now().toString(),
      date: document.getElementById('date-input').value,
      type: document.getElementById('type-select').value,
      category: document.getElementById('category-input').value,
      amount: parseFloat(document.getElementById('amount-input').value),
      description: document.getElementById('description-input').value,
      createdAt: new Date().toISOString()
      };
      const result = await window.dataSdk.create(transaction);
      submitBtn.disabled = false;
      submitText.textContent = 'Tambah Transaksi';
      submitSpinner.classList.add('hidden');
      if (result.isOk) {
      e.target.reset();
      document.getElementById('date-input').value = new Date().toISOString().split('T')[0];
      } else {
      alert('Gagal menyimpan transaksi. Silakan coba lagi.');
      }
      });
      document.getElementById('date-input').value = new Date().toISOString().split('T')[0];
      async function init() {
      if (window.dataSdk) {
      const initResult = await window.dataSdk.init(dataHandler);
      if (!initResult.isOk) {
      console.error("Gagal menginisialisasi Data SDK");
      }
      }
      if (window.elementSdk) {
      window.elementSdk.init({
      defaultConfig,
      onConfigChange,
      mapToCapabilities: (config) => ({
      recolorables: [],
      borderables: [],
      fontEditable: undefined,
      fontSizeable: undefined
      }),
      mapToEditPanelValues: (config) => new Map([
      ["app_title", config.app_title || defaultConfig.app_title],
      ["income_label", config.income_label || defaultConfig.income_label],
      ["expense_label", config.expense_label || defaultConfig.expense_label]
      ])
      });
      }
      }
